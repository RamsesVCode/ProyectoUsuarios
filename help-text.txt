--TABLA USUARIOS
CREATE TABLE USUARIOS(
ID INTEGER(255) PRIMARY KEY AUTO_INCREMENT,
NOMBRE CHAR(255) NOT NULL,
APELLIDOS CHAR(255) NOT NULL,
CORREO CHAR(255) NOT NULL UNIQUE,
PASS CHAR(255) NOT NULL,
FECHA_NAC DATE,
GENERO CHAR(1),
FECHA_REG DATE
)ENGINE=INNODB;
